# PacketTrace is NOT a template, so it needs compilation.
# NetworkNode and SimulationController ARE templates, so they are header-only.
add_library(network_lib STATIC
    packet_trace.cpp
)

target_include_directories(network_lib PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/..
    ${CMAKE_CURRENT_SOURCE_DIR}/../core
    ${CMAKE_CURRENT_SOURCE_DIR}/../mesh
)

target_link_libraries(network_lib PUBLIC
    mesh_lib
    nlohmann_json::nlohmann_json
)

set_target_properties(network_lib PROPERTIES
    POSITION_INDEPENDENT_CODE ON
    CXX_STANDARD 17
    CXX_STANDARD_REQUIRED ON
)

target_compile_features(network_lib PUBLIC cxx_std_17)

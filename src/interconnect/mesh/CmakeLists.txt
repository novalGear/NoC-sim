# src/interconnect/mesh/CMakeLists.txt
#=============================================================================
# Mesh topology library (Header-only / Interface)
#=============================================================================

add_library(mesh_lib INTERFACE)

target_include_directories(mesh_lib INTERFACE
    ${CMAKE_CURRENT_SOURCE_DIR}      # For #include "mesh.hpp"
    ${CMAKE_CURRENT_SOURCE_DIR}/..   # For #include "../core/router.hpp"
)

target_link_libraries(mesh_lib INTERFACE
    interconnect_core  # Depends on core
)

# C++ Standard requirements
target_compile_features(mesh_lib INTERFACE cxx_std_17)
